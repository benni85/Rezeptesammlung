\textbf{{\LARGE Lamm Satay}}%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%Seite 91 "the food of indonesia"

\hrulefill

Auf Java übliche Satay-Variante, benannt nach der Insel Madura 
\vspace*{\fill}
\begin{multicols}{2}	
Fleischspiesse:
\begin{itemize}
\item 750 g Lammfleisch %boneless lamb
\item 24 Bambusspiesse%bamboo skewers, soaked in water for 4 hours before using
\item Limettenspalte %Lime wedges, to serve
\end{itemize}
\vfill
\columnbreak
Marinade:
\begin{itemize}
\item 125 ml Süße indonesische Sojasauce (kecap manis)%sweet Indonesian soy sauce (kecap manis)
\item $\frac{1}{2}$ TL zerkleinerte Koriandersamen %teaspoon coriander seeds, crushed, or $\frac{1}{2}$ teaspoon ground coriander
\item $\frac{1}{4}$ TL frischgemahlener weisser Pfeffer%teaspoon ground white pepper
\item 1 TL frischgepresster Limetten- o. Zitronensaft%tablespoon freshly squeezed lime or lemon juice
\end{itemize}
\end{multicols}

\vspace{5mm}
\begin{center}
[Zutaten für 24 Satays, reicht für 4-6 Personen]
\end{center}
